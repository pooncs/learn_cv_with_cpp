cmake_minimum_required(VERSION 3.15)
project(ex06_inference_execution CXX)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
find_package(GTest REQUIRED)
add_executable(main src/main.cpp)
enable_testing()
add_executable(test_inference_execution tests/test_inference_execution.cpp)
target_link_libraries(test_inference_execution PRIVATE GTest::gtest_main)
add_test(NAME test_inference_execution COMMAND test_inference_execution)

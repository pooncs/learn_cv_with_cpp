cmake_minimum_required(VERSION 3.15)
project(ex05_input_buffers CXX)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
find_package(GTest REQUIRED)
add_executable(main src/main.cpp)
enable_testing()
add_executable(test_input_buffers tests/test_input_buffers.cpp)
target_link_libraries(test_input_buffers PRIVATE GTest::gtest_main)
add_test(NAME test_input_buffers COMMAND test_input_buffers)

cmake_minimum_required(VERSION 3.15)
project(ex08_postprocessing CXX)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
find_package(GTest REQUIRED)
add_executable(main src/main.cpp)
enable_testing()
add_executable(test_postprocessing tests/test_postprocessing.cpp)
target_link_libraries(test_postprocessing PRIVATE GTest::gtest_main)
add_test(NAME test_postprocessing COMMAND test_postprocessing)

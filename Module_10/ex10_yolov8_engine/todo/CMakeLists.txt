cmake_minimum_required(VERSION 3.15)
project(ex10_yolov8_engine CXX)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
find_package(GTest REQUIRED)
add_executable(main src/main.cpp)
enable_testing()
add_executable(test_yolov8_engine tests/test_yolov8_engine.cpp)
target_link_libraries(test_yolov8_engine PRIVATE GTest::gtest_main)
add_test(NAME test_yolov8_engine COMMAND test_yolov8_engine)

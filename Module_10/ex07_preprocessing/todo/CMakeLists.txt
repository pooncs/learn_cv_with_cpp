cmake_minimum_required(VERSION 3.15)
project(ex07_preprocessing CXX)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
find_package(GTest REQUIRED)
add_executable(main src/main.cpp)
enable_testing()
add_executable(test_preprocessing tests/test_preprocessing.cpp)
target_link_libraries(test_preprocessing PRIVATE GTest::gtest_main)
add_test(NAME test_preprocessing COMMAND test_preprocessing)

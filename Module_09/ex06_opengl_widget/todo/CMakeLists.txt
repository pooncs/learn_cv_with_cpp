cmake_minimum_required(VERSION 3.15)
project(ex06_opengl_widget CXX)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)
find_package(Qt6 COMPONENTS Widgets REQUIRED)
find_package(GTest REQUIRED)
add_executable(main src/main.cpp)
target_link_libraries(main PRIVATE Qt6::Widgets)
enable_testing()
add_executable(test_opengl_widget tests/test_opengl_widget.cpp)
target_link_libraries(test_opengl_widget PRIVATE GTest::gtest_main Qt6::Widgets)
add_test(NAME test_opengl_widget COMMAND test_opengl_widget)
